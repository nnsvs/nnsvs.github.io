from torch import nn

from nnsvs.base import BaseModel, PredictionType

from nnsvs.model import VariancePredictor
from nnsvs.util import init_weights



[docs]class RelDurMDN(BaseModel):
    """Relative duration prediction model based on MDN

    Instead of predicting phoneme durations directly, the model
    predicts relative phoneme durations per note.
    This model is inpired by :cite:t:`zhang2022visinger`.

    Args:
        in_dim (int): the dimension of the input
        hidden_dim (int): the dimension of the hidden state
        out_dim (int): the dimension of the output
        num_layers (int): the number of layers
        dropout (float): the dropout rate
        num_gaussians (int): the number of gaussians
        dim_wise (bool): whether to use dimension-wise or not
        init_type (str): the type of weight initialization
        in_note_dur_idx (int): the index of the input note duration
        in_note_dur_min (float): the minimum of the input note duration
        in_note_dur_max (float): the maximum of the input note duration
        out_dur_mean (float): the mean of the output note duration
        out_dur_scale (float): the scale of the output note duration
    """
    def __init__(
        self,
        in_dim,
        hidden_dim,
        out_dim,
        num_layers=1,
        dropout=0.0,
        num_gaussians=8,
        dim_wise=False,
        init_type="none",
        in_note_dur_idx=300,
        in_note_dur_min=1.0,
        in_note_dur_max=1.0,
        out_dur_mean=1.0,
        out_dur_scale=1.0,
    ):
        super(RelDurMDN, self).__init__()
        self.in_note_dur_idx = in_note_dur_idx
        self.in_note_dur_min = in_note_dur_min
        self.in_note_dur_max = in_note_dur_max
        self.out_dur_mean = out_dur_mean
        self.out_dur_scale = out_dur_scale

        model = [nn.Linear(in_dim, hidden_dim), nn.ReLU(), nn.Dropout(dropout)]
        if num_layers > 1:
            for _ in range(num_layers - 1):
                model += [nn.Linear(hidden_dim, hidden_dim), nn.ReLU(), nn.Dropout(dropout)]
        model += [
            MDNLayer(
                in_dim=hidden_dim,
                out_dim=out_dim,
                num_gaussians=num_gaussians,
                dim_wise=dim_wise,
            )
        ]
        self.model = nn.Sequential(*model)
        init_weights(self, init_type)

[docs]    def forward(self, x, lengths=None, y=None):
        """Forward step

        Args:
            x (torch.Tensor): the input tensor
            lengths (torch.Tensor): the lengths of the input tensor
            y (torch.Tensor): the optional target tensor

        Returns:
            torch.Tensor: the output tensor
        """
        log_pi, log_sigma, relative_mean_dur = self.model(x)

        # change time-resolution from 0.01 to 0.005 seconds
        dur_score = x[:, :, self.in_note_dur_idx].unsqueeze(-1) * 2
        dur_score_denorm = (
            dur_score * (self.in_note_dur_max - self.in_note_dur_min)
            + self.in_note_dur_min
        )

        # eq (8) in https://arxiv.org/abs/2110.08813
        pred_dur_mu_denorm = relative_mean_dur * dur_score_denorm.unsqueeze(-1)

        pred_dur_mu = (
            pred_dur_mu_denorm - self.out_dur_mean
        ) / self.out_dur_scale

        return log_pi, log_sigma, pred_dur_mu



[docs]    def inference(self, x, lengths=None):
        """Inference step

        Find the most likely mean and variance

        Args:
            x (torch.Tensor): the input tensor
            lengths (torch.Tensor): the lengths of the input tensor

        Returns:
            tuple: mean and variance of the output features
        """
        log_pi, log_sigma, mu = self(x, lengths)
        sigma, mu = mdn_get_most_probable_sigma_and_mu(log_pi, log_sigma, mu)
        return mu, sigma

    """Relative duration prediction model based on the VariancePredictor and MDN.

    References: :cite:t:`ren2020fastspeech`, :cite:t:`zhang2022visinger`.

    Args:
        in_dim (int): the dimension of the input
        out_dim (int): the dimension of the output
        num_layers (int): the number of layers
        hidden_dim (int): the dimension of the hidden state
        kernel_size (int): the kernel size of the convolutional layer
        dropout (float): the dropout rate
        num_gaussians (int): the number of gaussians
        dim_wise (bool): whether to use dimension-wise or not
        init_type (str): the type of weight initialization
        in_note_dur_idx (int): the index of the input note duration
        in_note_dur_min (float): the minimum of the input note duration
        in_note_dur_max (float): the maximum of the input note duration
        out_dur_mean (float): the mean of the output note duration
        out_dur_scale (float): the scale of the output note duration
    """
    def __init__(
        self,
        in_dim,
        out_dim,
        num_layers=5,
        hidden_dim=256,
        kernel_size=5,
        dropout=0.5,
        use_mdn=True,
        num_gaussians=1,
        dim_wise=False,
        init_type="none",
        in_note_dur_idx=300,
        in_note_dur_min=1.0,
        in_note_dur_max=1.0,
        out_dur_mean=1.0,
        out_dur_scale=1.0,
    ):
        super().__init__(
            in_dim=in_dim,
            out_dim=out_dim,
            num_layers=num_layers,
            hidden_dim=hidden_dim,
            kernel_size=kernel_size,
            dropout=dropout,
            init_type=init_type,
            use_mdn=use_mdn,
            num_gaussians=num_gaussians,
            dim_wise=dim_wise,
        )
        self.in_note_dur_idx = in_note_dur_idx
        self.in_note_dur_min = in_note_dur_min
        self.in_note_dur_max = in_note_dur_max
        self.out_dur_mean = out_dur_mean
        self.out_dur_scale = out_dur_scale

        if self.use_mdn:
            self.log_sigma2 = nn.Linear(out_dim * num_gaussians, out_dim * num_gaussians)
        else:
            self.log_sigma2 = None



[docs]    def forward(self, x, lengths=None, y=None):
        """Forward step

        Args:
            x (torch.Tensor): the input tensor
            lengths (torch.Tensor): the lengths of the input tensor
            y (torch.Tensor): the optional target tensor

        Returns:
            torch.Tensor: the output tensor
        """
        out = super().forward(x, lengths, y)
        if self.use_mdn:
            log_pi, log_sigma, relative_mean_dur = out
        else:
            relative_mean_dur = out

        # change time-resolution from 0.01 to 0.005 seconds
        dur_score = x[:, :, self.in_note_dur_idx].unsqueeze(-1) * 2
        dur_score_denorm = (
            dur_score * (self.in_note_dur_max - self.in_note_dur_min)
            + self.in_note_dur_min
        )

        if self.use_mdn:
            dur_score_denorm = dur_score_denorm.unsqueeze(-1)

        # eq (8) in https://arxiv.org/abs/2110.08813
        pred_dur_mu_denorm = relative_mean_dur * dur_score_denorm

        pred_dur_mu = (
            pred_dur_mu_denorm - self.out_dur_mean
        ) / self.out_dur_scale

        if self.use_mdn:
            return log_pi, log_sigma, pred_dur_mu
        else:
            return pred_dur_mu




[docs]    def inference(self, x, lengths=None):
        """Inference step

        Args:
            x (torch.Tensor): input features
            lengths (torch.Tensor): lengths of input features

        Returns:
            tuple: (mu, sigma) if use_mdn, (output, ) otherwise
        """
        if self.use_mdn:
            log_pi, log_sigma, mu = self(x, lengths)
            sigma, mu = mdn_get_most_probable_sigma_and_mu(log_pi, log_sigma, mu)
            return mu, sigma
        else:
            return self(x, lengths)
